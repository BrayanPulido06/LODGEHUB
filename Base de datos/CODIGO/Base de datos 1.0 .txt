create table huespedes ( num_Documento int (10) not null,
ti_documento varchar (10) not null,
mot_viaje varchar (20) not null,
correo varchar (35) not null, 
sexo varchar (15) not null,
est_civil (15) not null,
nombre (70) not null,

primary key (num_Documento));



create PQRS (id_pqrs int (10) not null,
categoria varchar (15) not null,
fecha date not null,
estado varchar (15) not null,
niv_urgencia varchar (15) not null,
descripcion varchar (100) not null,

primary key (id_pqrs));
